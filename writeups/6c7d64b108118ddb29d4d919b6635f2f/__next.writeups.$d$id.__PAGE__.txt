def lagrange_interpolation(coords, p):
    t = len(coords)
    Zp = Integers(p)
    R.<X> = Zp[]
    x_tup, y_tup = zip(*coords)

    polys = []
    for i in [0, .., t-1]:
        polys.append(1)
        for j in [0, .., t-1]:
            if (j == i):
                polys[i] = polys[i] * y_tup[i]
            else:
                polys[i] = polys[i] * (X - x_tup[j]) / (x_tup[i] - x_tup[j])
    PprimeX = 0
    for i in [0, .., t-1]:
        PprimeX = PprimeX + polys[i]
    return PprimeX
